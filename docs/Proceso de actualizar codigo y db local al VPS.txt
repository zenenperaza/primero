1. En tu PC local: generar dump
cd ~/primero/docker

docker compose -p primero exec postgres sh -c "pg_dump -U primero --encoding=UTF8 --no-owner --no-acl primero_production > /tmp/db_local.sql"
docker compose -p primero cp postgres:/tmp/db_local.sql ./db_local.sql


Sube db_local.sql al VPS en:

/home/zenenperaza/db_local.sql

cd ~/primero/docker

docker compose -p primero stop application worker

docker compose -p primero exec postgres psql -U primero -d primero_production -c "DROP SCHEMA public CASCADE; CREATE SCHEMA public;"

docker compose -p primero exec -T postgres psql -U primero -d primero_production -v ON_ERROR_STOP=1 < ~/db_local.sql

docker compose -p primero start application worker
docker compose -p primero restart
